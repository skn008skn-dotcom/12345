#include <iostream>
#include <random>
#include <clocale>

int main()
{
    setlocale(LC_ALL, "Russian"); 

    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> distr(0, 100);

    int загаданное = distr(gen);
    int попытка;

    std::cout << "Угадай число от 0 до 100!\n";

    while (true)
    {
        std::cin >> попытка;
        if (попытка == загаданное)
        {
            std::cout << "Угадал!\n";
            break;
        }
        else if (попытка < загаданное)
        {
            std::cout << "Надо больше\n";
        }
        else
        {
            std::cout << "Надо меньше\n";
        }
    }

    return 0;
